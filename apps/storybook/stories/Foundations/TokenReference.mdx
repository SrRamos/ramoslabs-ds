import { useEffect } from 'react';

<Meta title="Foundations/Token Reference" />

export const CopySetup = () => {
  useEffect(() => {
    const handleClick = (e) => {
      const token = e.target.closest('[data-token]');
      if (token) {
        const tokenValue = token.dataset.token;
        const textToCopy = `var(${tokenValue})`;
        navigator.clipboard.writeText(textToCopy).then(() => {
          // Show toast
          const toast = document.getElementById('copy-toast');
          const toastText = document.getElementById('copy-toast-text');
          if (toast && toastText) {
            toastText.textContent = `Copied: ${textToCopy}`;
            toast.classList.add('show');
            setTimeout(() => {
              toast.classList.remove('show');
            }, 2000);
          }
          // Visual feedback on token
          token.classList.add('copied');
          setTimeout(() => token.classList.remove('copied'), 1000);
        });
      }
    };
    document.addEventListener('click', handleClick);
    return () => document.removeEventListener('click', handleClick);
  }, []);
  return null;
};
