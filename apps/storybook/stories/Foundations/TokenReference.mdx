export const CopySetup = () => {
  useEffect(() => {
    const handleClick = (e) => {
      const token = e.target.closest('[data-token]');
      if (token) {
        const tokenValue = token.dataset.token;
        const textToCopy = `var(${tokenValue})`;
        navigator.clipboard.writeText(textToCopy).then(() => {
          const toast = document.getElementById('copy-toast');
          const toastText = document.getElementById('copy-toast-text');
          if (toast && toastText) {
            toastText.textContent = `Copied: ${textToCopy}`;
            toast.classList.add('show');
            setTimeout(() => {
              toast.classList.remove('show');
            }, 2000);
          }
          token.classList.add('copied');
          setTimeout(() => token.classList.remove('copied'), 1000);
        });
      }
    };
    document.addEventListener('click', handleClick);
    return () => document.removeEventListener('click', handleClick);
  }, []);
  return null;
};

<CopySetup />

  <h3 style={{ fontFamily: "'Red Hat Display', sans-serif", fontSize: '1rem', color: '#0f172a', marginTop: '1.5rem', marginBottom: '0.5rem' }}>How to consume a token</h3>

    Reference the variable, never the literal. Reach for the semantic tier first, because it carries the meaning and stays stable across themes. Drop to a primitive only when no semantic token fits, and if you find yourself wanting a raw hex, that is the signal a semantic token is missing, not that you should hardcode.
  </p>

  <h3 style={{ fontFamily: "'Red Hat Display', sans-serif", fontSize: '1rem', color: '#0f172a', marginTop: '2rem', marginBottom: '0.5rem' }}>How to name a new token</h3>
  <div className="decision">
    <div className="decision-step">
      <div className="decision-text">Pick the tier. Is it a raw value with no role? It belongs in <strong>primitive</strong>, named by description, for example a new palette shade. Does it carry meaning? It belongs in <strong>semantic</strong>, named by role.</div>
    </div>
    <div className="decision-step">
      <div className="decision-text">Name a semantic token role first, variant second: <strong>category, role, variant</strong>. So <code>color</code>, <code>text</code>, <code>secondary</code> becomes <code>--color-text-secondary</code>. Never encode the hue in the name.</div>
    </div>
    <div className="decision-step">
      <div className="decision-text">Alias, do not copy. A semantic token points at a primitive with <code>{`{color.slate.600}`}</code>, so the value has one home. If two tokens share a value, both alias the same primitive.</div>
    </div>
    <div className="decision-step">
      <div className="decision-text">Add a <code>$description</code> when the role is not obvious from the name. It becomes the comment in the generated CSS and the note in this reference.</div>
    </div>
  </div>

  <h3 style={{ fontFamily: "'Red Hat Display', sans-serif", fontSize: '1rem', color: '#0f172a', marginTop: '2rem', marginBottom: '0.5rem' }}>How a change propagates</h3>
  <p className="token-ref-lede">
    A change starts in the JSON and ends on the screen without a single manual edit downstream. Edit the <code>$value</code> in <code>packages/tokens/src/tokens/</code>, run the Style Dictionary build, and the generated <code>tokens.css</code> and <code>tokens.ts</code> update together. Because every consumer references the variable, the new value reaches every product surface at once. Never edit the generated files: they carry a do-not-edit banner and the next build overwrites them.
  </p>

  <div className="callout">
    <div className="callout-title">One source, many outputs</div>
    <div className="callout-body">
      The JSON is the only file a person edits. <code>dist/tokens.css</code>, <code>dist/tokens.ts</code>, and <code>dist/tokens.json</code> are all build output. If a value looks wrong on screen, fix it in the source token and rebuild, never in the CSS.
    </div>
  </div>

  <div className="rules-grid">
    <div className="rules-col do">
      <div className="rules-col-title"><span className="rules-badge">+</span> Do</div>
      <ul className="rules-list">
        <li>Consume <code>var(--token)</code> references, and prefer the semantic tier.</li>
        <li>Author new tokens in the JSON source, then rebuild.</li>
        <li>Name semantic tokens by role, category first, hue never.</li>
        <li>Alias a primitive instead of repeating a hex.</li>
        <li>Add a <code>$description</code> for any non-obvious role.</li>
      </ul>
    </div>
    <div className="rules-col dont">
      <div className="rules-col-title"><span className="rules-badge">x</span> Don't</div>
      <ul className="rules-list">
        <li>Hardcode a hex, px, or raw integer that a token already covers.</li>
        <li>Edit <code>dist/tokens.css</code> or <code>dist/tokens.ts</code> by hand.</li>
        <li>Reach for a primitive when a semantic token fits.</li>
        <li>Encode a color name into a semantic token.</li>
        <li>Duplicate a value across tokens instead of aliasing.</li>
      </ul>
    </div>
  </div>
